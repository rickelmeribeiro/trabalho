{
    $conn = conectar();

        $conn->beginTransaction();

        $stmt = $conn->prepare("DELETE FROM nome da tabela");
        $stmt->execute();
        $conn->commit();

        if ($stmt->rowCount() > 0) {
            return $stmt->fetchAll(PDO::FETCH_OBJ);
